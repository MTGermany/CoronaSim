\visible<3-7>{\placebox{0.75}{0.83}{Odds ratio}}
\visible<3-7>{\placebox{0.75}{0.75}{\parbox{0.3\textwidth}{
\maindm{y=\frac{p}{1-p}}
}}}

\visible<4-7>{\placebox{0.755}{0.65}{\parbox{0.2\textwidth}{
\maindm{y=y_0e^{r_yt}}
}}}

\visible<5-7>{\placebox{0.81}{0.55}{Estimation growth rate}}

\visible<5-7>{\placebox{0.82}{0.45}{\parbox{0.3\textwidth}{
\maindm{r_y \approx \frac{\ln y(t_2)-\ln y(t_1)}{t_2-t_1}}
}}}

\visible<6-7>{\placebox{0.82}{0.34}{Estimation B.1.1.7 share}}

\visible<6-7>{\placebox{0.755}{0.25}{\parbox{0.2\textwidth}{
\maindm{p=\frac{y}{1+y}}
}}}

\visible<6-7>{\placebox{0.813}{0.16}{\parbox{0.3\textwidth}{
\maindm{\hspace*{1em}=\frac{p_0e^{r_yt}}{1+p_0\left(e^{r_yt}-1\right)}}
}}}

\visible<8->{\placebox{0.81}{0.66}{$\tau$: generation time}}

\visible<8->{\placebox{0.81}{0.60}{\parbox{0.2\textwidth}{
\maindm{R_0\sup{mut}/R_0\sup{wild} = \tau r_y+1}
}}}

\visible<9->{\placebox{0.81}{0.50}{Effective $R_0$:}}

\visible<9->{\placebox{0.83}{0.44}{\parbox{0.3\textwidth}{
\maindm{R_0=(1-p) R_0\sup{wild} + pR_0\sup{mut}}
}}}

\visible<10->{\placebox{0.83}{0.28}{\parbox{0.3\textwidth}{
\maindm{\Rightarrow \ R_0\sup{wild}, R_0\sup{mut} \ \text{known}}
}}}

\visible<10->{\placebox{0.836}{0.20}{\parbox{0.3\textwidth}{
\maindm{\Rightarrow \ \text{forecast} \ R_0(t)}
}}}

\frame{\frametitle{1.2 Modelling the age-specific infection fatality
    rate (IFR)} 
%##############################################################

\placebox{0.30}{0.42}{
  \figSimple{0.68\textwidth}{figsCorona/natureAgeSpecificIFR.png}}

\placebox{0.41}{0.275}
  {\scriptsize{M. O'Driscoll et. al., Nature \textbf{590}, pp. 140 (2021)}}

\placebox{0.80}{0.46}{\parbox{0.40\textwidth}{
\bi
\item The IFR exhibits a nearly exponential growth with the age
\item The same is true for the case fatality rate (CFR) and for the
  total one-year fatality
\item The gender difference is weak (corresponds to $\approx$ 5 years) 
\pause \item The infection probability exhibts a much weaker age dependency
\item \red{$\Rightarrow$ leave infection spread model unchanged}
  (enables interactive just-in-time calibration) and
  calculate the IFR in age strata afterwards
\ei
}}
